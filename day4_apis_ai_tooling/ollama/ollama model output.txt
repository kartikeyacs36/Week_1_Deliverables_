### Step 5: Create the Main Application File

In `app/main.py`, we set up FastAPI and configure SQLAlchemy.

**main.py**

```python
from fastapi import FastAPI, Depends, HTTPException, status
from sqlalchemy.orm import Session
from app.models.item_model import ItemBase, ItemCreate, ItemUpdate, Item
from app.database import get_db

app = FastAPI()

def get_db():
    try:
        db = engine.connect()
        yield db
    finally:
        db.close()

@app.post("/items/", response_model=Item)
async def create_item(item: ItemCreate, db: Session = Depends(get_db)):
    new_item = ItemBase(**item.dict())
    existing_item = db.query(Item).filter_by(name=new_item.name).first()
    
    if existing_item is None:
        item_dict = item.dict()
        item_dict['id'] = 1
        new_item = ItemCreate(**item_dict)
        db.add(new_item)
        db.commit()
        return new_item
    else:
        raise HTTPException(status_code=409, detail="Item already exists")

@app.get("/items/{item_id}", response_model=Item)
async def read_item(item_id: int, db: Session = Depends(get_db)):
    item = db.query(Item).filter_by(id=item_id).first()
    
    if not item:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Item not found")
    return item

@app.put("/items/{item_id}", response_model=Item)
async def update_item(item_id: int, updated_item: ItemUpdate, db: Session = Depends(get_db)):
    existing_item = db.query(Item).filter_by(id=item_id).first()
    
    if not existing_item:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Item not found")
    
    item_dict = updated_item.dict(exclude_unset=True)
    existing_item_dict = existing_item.__dict__
    for key, value in item_dict.items():
        setattr(existing_item, key, value)
    
    db.commit()
    return existing_item

@app.delete("/items/{item_id}", response_model=Item)
async def delete_item(item_id: int, db: Session = Depends(get_db)):
    existing_item = db.query(Item).filter_by(id=item_id).first()
    
    if not existing_item:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Item not found")
    
    item_dict = existing_item.__dict__
    del item_dict['id']
    db.delete(existing_item)
    db.commit()
    return item_dict
```

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session
from app.database.base_class import Base
from app.models.task_model import get_task_model, Task
from app.schemas.task_schema import TaskCreate, TaskUpdate, Task
from app.utils.helpers import get_db

router = APIRouter()

@router.post("/tasks/", response_model=Task)
async def create_task(task: TaskCreate, db: Session = Depends(get_db)):
    task_obj = TaskCreate(**task.dict())
    new_task = await get_task_model().create(db=db, obj_in=task_obj)
    return new_task

@router.get("/tasks/{task_id}", response_model=Task)

@router.put("/tasks/{task_id}", response_model=Task)
async def update_task(task_id: int, task_update: TaskUpdate, db: Session = Depends(get_db)):
    task = await get_task_model().get_by_id(db=db, id=task_id)
    if not task:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
    
    updated_task = await get_task_model().update(db=db, db_obj=task, obj_in=task_update)
    return updated_task

@router.delete("/tasks/{task_id}")
async def delete_task(task_id: int, db: Session = Depends(get_db)):

    if not task:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
    
    await get_task_model().delete(db=db, obj=task)
    return {"message": "Task deleted successfully"}
